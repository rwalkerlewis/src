 39  # 
 40  # def make_reservoir (  memsize=100,            transpose memory size 
 41  #                       private=None,           data server parameters 
 42  #                       grid_par=None,          grid parameters 
 43  #                       geo_par=None,           geometrical parameters 
 44  #                       sand_par=None,          sand fraction parameters 
 45  #                       bk_noise_par=None,      background noise parameters 
 46  #                       sd_noise_par=None,      sand noise parameters 
 47  #                       taper_par=None):        taper parameters 
 48  # 
 49  # def make_overburden ( memsize=100,            transpose memory size 
 50  #                       grid_par=None,          grid parameters 
 51  #                       noise_par=None,         background noise parameters 
 52  #                       taper_par=None):        taper parameters 
 53  # 
 54  # Grid parameters 
 55  # 
 56  # The z direction parameters may be different for the reservoir and 
 57  # overburden, but for seismic modeling the x and y direction parameters 
 58  # should be the same. 
 59  # 
 60  # grid_par = {  'nx', 'dx', 'ox',   x grid parameters 
 61  #               'ny', 'dy', 'oy',   y grid parameters 
 62  #               'nz', 'dz', 'oz',   z grid parameters 
 63  #               'xy_pad'            x & y direction padding 
 64  #            } 
 65  # 
 66  # Notes on specification: 
 67  # 
 68  # This current version of channels.py interpolates 15 2D arrays of 
 69  # channel position and skew from Mathematica-exported files to your 
 70  # desired grid.  The original files were generated in Mathematica on 
 71  # the following grid: 
 72  # 
 73  # n1=181   d1=30m   o1=0m  
 74  # n2=101   d2=30m   o2=0m           
 75  # 
 76  # The Mathematica grid spans: 
 77  # 
 78  # 0m <= x <= 5400m  and  0m <= y <= 3000m 
 79  # 
 80  # Your grid should be within this range to avoid extrapolation on the 
 81  # edges.  These grid parameters set the location and size of your grid 
 82  # relative to the Mathematica arrays.  After the channels are created 
 83  # optional x and y padding is added, increasing nx and ny to values 
 84  # larger than those specified here (see padding parameter below), and 
 85  # the grid origin is reset to zero. 
 86  # 
 87  # Future versions of channels.py may generate position and skew 
 88  # internally, avoiding the interpolation-extrapolation issue. 
 89  # 
 90  # The following example spans: 
 91  # 
 92  # 0m <= x <= (nx-1)*dx = 5385m  and  0m <= y <= (ny-1)*dy = 2985m 
 93  # 
 94  # nx=360   dx=15m   ox=0m          
 95  # ny=200   dy=15m   oy=0m           
 96  # 
 97  # The z coordinate measures height (not depth) above the bottom of the 
 98  # first channel.  The top of the last channel is at 110m. 
 99  # 
100  # There are two sets of vertical grid parameters, one for a reservoir 
101  # grid that contains the channels, and another set for a grid of 
102  # overburden with only background material with it's noise, but no 
103  # channels.  The two grids should probably have the same x and y grid 
104  # parameters, but they may have different z direction grid size, 
105  # spacing, and location. 
106  # 
107  # You can choose any vertical grid.  Choosing a reservoir grid smaller 
108  # than the channel range will produce a grid that clips some of the 
109  # channels. Choosing a larger grid will generate extra "pad" volume 
110  # above or below the channels.  After the channels are created the grid 
111  # origin is reset to zero.  The overburden grid will have no channels, 
112  # even if you choose it's location within the zone where the channels 
113  # exist. 
114  # 
115  # The following example creates a vertical reservoir grid that spans 
116  # -20m <= z <= 129m, with a 20m pad below the bottom channel, and a 19m 
117  # pad above the top channel: 
118  # 
119  # nz=150 dz=1m oz=-20m 
120  # 
121  # Additional padding in the x and y directions: xy_pad 
122  # 
123  # This parameter controls the width (in cells) for additional all-shale 
124  # padding to be added to all four horizontal sides of the model after 
125  # the channels are constructed.  After the pad is added, the x and y 
126  # origin is reset to zero. 
127  # 
128  # Channel geometrical properties 
129  # 
130  # geo_par = {                   amalgamated sand geometrical properties 
131  #               'as_width'      width at channel bottom 
132  #               'as_shift'      lateral shift at channel max bend 
133  #               'as_height0'    height at channel inflection 
134  #               'as_height1'    height at channel max bend 
135  #               'as_shape'      cross-section shape factor 
136  # 
137  #                               bar drape geometrical properties 
138  #               'bd_depth'      bar drape profile depth 
139  #               'bd_zshift'     bar drape profile z shift 
140  # 
141  #                               margin drape geometrical properties 
142  #               'md_depth'      margin drape profile depth 
143  #               'md_zshift'     margin drape profile z shift 
144  #           } 
145  #  
146  # Sand fraction parameters  
147  # 
148  # sand_par = {  'bd_sand'   bypass drape sand fraction (sf) 
149  #               'md_sand'   margin drape sf 
150  #               'as_sand'   amalgamated sand sf 
151  #               'na_sand0'  non-amalgamated sand sf at channel top 
152  #               'na_sand1'  non-amalgamated sand sf at amalgamated sand contact 
153  #            } 
154  # 
155  # Porosity noise parameters 
156  # 
157  # The reservoir requires two of noise parameters, bk_noise_par and 
158  # sd_noise_par, for the background and the sand.  A third set of noise 
159  # parameters may be used for the overburden, although it commonly is 
160  # the same as bk_noise_par.  See rfield.py for notes on the covariance 
161  # parameters for random field generation. 
162  # 
163  # These parameters are used to generate random fields of noise to be 
164  # added to the background and sand porosity.  The same noise propogates 
165  # through the subsequent generation of the rho, vp, and vs grids. 
166  # 
167  # The code will make one realizataion for the background noise, and 15 
168  # different realizations for the channels, one for each of the 15 
169  # channels in the model.  The 15 sand realizatins are all made with the 
170  # same set of parameters.  The random number seeds are preset.  A small 
171  # coding change would allow for user setting of the random number seeds 
172  # for different realizations of the entire model. 
173  # 
174  # noise_par = { 'taper_switch'  covariance taper switch 
175  #               'std_dev'       porosity noise standard devietion 
176  #               'alpha',        covariance shape parameter 
177  #               'oriu',         covariance range orientation vectors 
178  #               'oriv',  
179  #               'oriw', 
180  #               'ru',           covariance range parameters 
181  #               'rv',    
182  #               'rw'} 
183  # 
184  # Taper parameters 
185  # 
186  # A linear taper is applied, if desired, to the rock properties (phi, 
187  # rho, vp, and vs) at the top and bottom of each of the reservoir and 
188  # overburden grids.  However, it probably makes most sense to use only 
189  # a top taper in the overburden, and only a bottom taper in the 
190  # reservoir. 
191  #  
192  # The properties are linearly interpolated between those generated by 
193  # the channel model, and a set of constants for the top (or bottom) of 
194  # the grid. The weighting is computed to produce the constant values at 
195  # the top (or bottom) of the taper zone, the channel model output at 
196  # the bottom (or top) of the taper zone and below (or above), and 
197  # linear weighting with depth in between. 
198  # 
199  # If a taper thickness is set to zero no taper will be applied in the 
200  # corresponding taper zone. 
201  # 
202  # taper_par = {             top taper parameters 
203  #               'top_h'     thickness (m) 
204  #               'top_phi'   porosity (fraction) 
205  #               'top_rho'   density (gm/cc) 
206  #               'top_vp'    Vp (m/s) 
207  #               'top_vs'    Vs (m/s) 
208  # 
209  #                           bottom taper parameters 
210  #               'bot_h'     thickness (m) 
211  #               'bot_phi'   porosity (fraction) 
212  #               'bot_rho'   density (gm/cc) 
213  #               'bot_vp'    Vp (m/s) 
214  #               'bot_vs'    Vs (m/s) 
215  #             }
