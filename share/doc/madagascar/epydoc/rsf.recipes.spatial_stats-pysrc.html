 30  # Fast spatial statistics computation with FFTs: 
 31  #   semi-variogram 
 32  #   general relative semi-variogram 
 33  #   covariance 
 34  #   correlogram 
 35  # 
 36  # Definitions for these spatial statistics can be found in: 
 37  # 
 38  # Deutsch, C. V., and Journel, A. G., 1998, GSLIB-Geostatistical 
 39  #   software library and user's guide: New York, Oxford University Press, 
 40  #   pp 44-45. 
 41  # 
 42  # Algorithm inspired from: 
 43  # 
 44  # Marcotte, D., 1996, Fast variogram computation with FFT,  
 45  #   Computers and Geosciences, v. 22, n. 10, pp. 1175-1186. 
 46  #  
 47   
 48  #  
 49  # Input parameter explanations. 
 50  # 
 51   
 52  # 
 53  # The input parameters are loaded into a dictionary 'par': 
 54  # 
 55  # par = {   'ind', 'val':       Indicator and data filenames 
 56  #           'nx', 'ny', 'nz':   Array sizes 
 57  #           'dx', 'dy', 'dz':   Cell sizes 
 58  #       } 
 59  # 
 60   
 61  # 
 62  # Value and indicator data files: par['ind','val'] 
 63  # 
 64  # These functions will compute a spatial statistics grid from the data 
 65  # values stored in the file named in par['val'].  The data must be 
 66  # stored in a regular 1, 2, or 3 dimensional array with array and cell 
 67  # sizes specified elsewhere in the dictionary "par" (see below).  
 68  # However, missing values are allowed. 
 69  # 
 70  # The array named in par['ind'] should contain only zeros and ones 
 71  # indicating the absence (zero) or presence (one) of data in the 
 72  # corresponding value array.  The value array should have zero values 
 73  # in locations where data is missing.  The value and indicator arrays 
 74  # should share the same dimensions. 
 75  # 
 76   
 77  # 
 78  # Array sizes: par['nx','ny','nz'] 
 79  # 
 80  # The code works for 1D, 2D, or 3D data grids.  For 2D or 1D set one 
 81  # or two of the array dimensions to 1. 
 82  # 
 83  # Use only even values (or 1) for nx, ny, and nz.  I'm not convinced 
 84  # the code is working correctly for odd values. 
 85  # 
 86  # The FFTs will be most efficient if the array sizes can be factored 
 87  # into combinations of 2, 3, 4, and 5.  Nevertheless, the FFTs may 
 88  # still be fast if a remaining factor bigger than 5 is not too large. 
 89  # 
 90  # Use integers. 
 91  # 
 92   
 93  # 
 94  # Cell sizes: par['dx','dy','dz'] 
 95  # 
 96  # You may use integers or floats, conversion to float is done for 
 97  # you when necessary. 
 98  # 
 99

101  # End of input parameter explanations.

155  # 
156  # Set up forward and inverse FFT rules  
157  # 
158   
159      fft_rule =      ''' 
160                      pad n1=%(nx_pad)d n2=%(ny_pad)d n3=%(nz_pad)d |  
161                      rtoc |  
162                      fft3 axis=1 pad=1 | 
163                      fft3 axis=2 pad=1 | 
164                      fft3 axis=3 pad=1 
165                      ''' % par 
166   
167      inv_fft_rule =  ''' 
168                      fft3 axis=3 pad=1 inv=y |  
169                      fft3 axis=2 pad=1 inv=y |  
170                      fft3 axis=1 pad=1 inv=y |  
171                      real 
172                      ''' 
173                       
174      rotate_rule =   'rotate rot1=%(nx)d rot2=%(ny)d rot3=%(nz)d ' % par 
175       
176      window_rule =   ''' 
177                      window f1=%(nx_half)d n1=%(nx_aug)d 
178                             f2=%(ny_half)d n2=%(ny_aug)d 
179                             f3=%(nz_half)d n3=%(nz_aug)d |  
180                      put o1=%(ox)g o2=%(oy)g o3=%(oz)g 
181                          d1=%(dx)g d2=%(dy)g d3=%(dz)g 
182                      ''' % par 
183                       
184      inv_rule = inv_fft_rule + '|' + rotate_rule + '|' + window_rule
