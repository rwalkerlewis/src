dt= 		time step

file fft= 		auxiliary input file name
 
file left= 		auxiliary output file name
 
nh=
 
npk=20 		maximum rank
 
seed=time(NULL
 
sub=true 		if subtract one
 
tol=1.e-4 		tolerance
